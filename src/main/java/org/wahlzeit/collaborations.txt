/**
 *  Collaboration between BookPhoto and Book
 */
public collaboration BookPhoto {

	public role BookPhoto {
		public Book getBook();


		public void setBook(Book);
	}


	public role Book {
		// no methods
	}
}

//Bindings
public class BookPhoto extends Photo binds Book.BookPhoto {
	
	public Book getBook() {
		return Book.BookPhoto.getBook();
	}

	public void setBook(Book book) {
		Book.BookPhoto.setBook(book);
	}
}


/**
 *  Collaboration between Book and BookType
 */
public collaboration Book {

	public role Book {
		public BookType getType();
	}


	public role BookType {
		public Book createInstance();
		
		public boolean hasInstance(Book):
	}
}

//Bindings
public class Book extends DataObject binds Book.Book {
	public BookType getType() {
		return Book.Book.getType();
	}
}

public class BookType extends DataObject binds Book.BookType {
	public Book createInstance() {
		return Book.BookType.createInstance();
	}
	
	public boolean hasInstance(Book book) {
		return Book.BookType.hasInstance(book);
	}
}
